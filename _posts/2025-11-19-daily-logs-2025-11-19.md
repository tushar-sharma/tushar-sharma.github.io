

```java

import static java.lang.System.*;

// Enum with a field and constructor
enum S1 {
    SHOP_EAST("HAIKU"); // Each constant can have its own value

    private final String code; // Field to store extra data

    // Constructor to set the value
    S1(String code) {
        this.code = code;
    }

    // Method to access the value
    public String getCode() {
        return code;
    }
}

// Simple enum with just names
enum S2 {
   FOO,
   BAR
}

public class Test {

    public static void main(String... args) {
        // Using S1 enum
        S1 s1 = S1.SHOP_EAST;
        out.println("S1 value: " + s1);           // Prints the enum constant name
        out.println("S1 code: " + s1.getCode());  // Prints the associated code

        // Using S2 enum
        for (S2 s2 : S2.values()) {
            out.println("S2 value: " + s2);       // Prints each enum constant name
        }
    }
}
```
